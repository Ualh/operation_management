def opt_production_plan(data):

    # Create the mip solver with the SCIP backend.
    solver = pywraplp.Solver.CreateSolver('SCIP')

    if not solver:
        return

    # Decision variables
    setup = []
    production = []
    inventory = []
    for product in range(len(data.demands)):
        setup.append([solver.BoolVar(f'setup_{product}_{day}') for day in range(len(data.days))])
        production.append([solver.NumVar(0, solver.infinity(), f'production_{product}_{day}') for day in range(len(data.days))])
        inventory.append([solver.NumVar(0, solver.infinity(), f'inventory_{product}_{day}') for day in range(len(data.days))])

    # Constraints
    for product in range(len(data.demands)):
        solver.Add(data.first_inventory[product] + production[product][0] == data.demands[product][0] + inventory[product][0])
        for day in range(1, len(data.days)):
            if day - data.lead_time[product] >= 0:
                solver.Add(inventory[product][day-1] + production[product][day - data.lead_time[product]] == \
                          (data.demands[product][day] + sum(data.parts[unit][product] * production[unit][day] for unit in range(len(data.demands)))) + inventory[product][day])
            else:
                solver.Add(inventory[product][day-1] == (data.demands[product][day] + sum(data.parts[unit][product] * production[unit][day] for unit in range(len(data.demands)))) + inventory[product][day])

            solver.Add(production[product][day] <= solver.infinity() * setup[product][day])

    # Objective function
    objective = solver.Sum(setup[product][day] * data.setup_cost[product] +
                           production[product][day] * data.production_cost[product] +
                           inventory[product][day] * data.inventory_cost[product]
                           for product in range(len(data.demands))
                           for day in range(len(data.days)))

    solver.Minimize(objective)

    # Solving the model
    status = solver.Solve()

    if status == pywraplp.Solver.OPTIMAL:
        PRODUCTION_PLAN = {}
        for product in range(len(data.demands)):
            PRODUCTION_PLAN_temp = {}
            for day_idx, day in enumerate(data.days):
                PRODUCTION_PLAN_temp[day] = {
                    'product': production[product][day_idx].solution_value(),
                    'setup': setup[product][day_idx].solution_value(),
                    'inventory': inventory[product][day_idx].solution_value()
                }
            PRODUCTION_PLAN[product] = PRODUCTION_PLAN_temp
        return PRODUCTION_PLAN, objective.solution_value()
    else:
        print('The problem does not have an optimal solution.')
        return None
